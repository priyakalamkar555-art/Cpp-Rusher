 8. Reverse a Number (using do-while loop)



#include <iostream>
using namespace std;

int main() {
    int num, rev = 0;
    cout << "Enter number: ";
    cin >> num;

    do {
        int digit = num % 10;   // get last digit
        rev = rev * 10 + digit; // build reversed number
        num /= 10;              // remove last digit
    } while (num != 0);

    cout << "Reversed number = " << rev;
    return 0;
}
